--- Clone ---

$ git clone http://llvm.org/git/llvm.git

$ cd llvm/tools
$ git clone http://llvm.org/git/clang.git

$ cd llvm/projects
$ git clone http://llvm.org/git/compiler-rt.git

--- Build ---

$ mkdir build/
$ cd build/
$ cmake -G Ninja -DCMAKE_EXPORT_COMPILE_COMMANDS=ON ..
# Or to build with Clang:
$ CXX=clang++ CC=clang cmake -G Ninja -DCMAKE_EXPORT_COMPILE_COMMANDS=ON -DCMAKE_BUILD_TYPE=DEBUG ..
$ cd path/to/llvm/sources
$ ninja

--- With ccache ---
$ CXX="ccache clang++" CC="ccache clang" CCACHE_CPP2=yes cmake -G Ninja -DCMAKE_EXPORT_COMPILE_COMMANDS=ON -DCMAKE_BUILD_TYPE=DEBUG ..
